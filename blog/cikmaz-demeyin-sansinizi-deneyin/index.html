(define (sayisal)
  (display "Oynamak istediginiz kolan sayisini giriniz: \n")
  (let ((kolon (read)))
    (do ((i 1 (+ i 1))) ((> i kolon))
      (let loop ((lst (list (+ (random 49) 1))) 
                   (sayi (+ 1 (random 49))))
        (if (= (length lst) 6)
            (printf "~a. kolon: ~a\n" i (sort lst < ))
            (if (member sayi lst)
                (loop lst (+ (random 49) 1))
                (loop (append lst (list sayi)) 
                       (+ (random 49) 1))))))))
(sayisal);burası kodun çalıştırılması için.
